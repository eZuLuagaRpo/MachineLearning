from PIL import Image

def fade_recursivo(img1, img2, i=0, n=30):
    if i == n:
        return []
    a = i / (n - 1)
    f = (img1 * (1 - a) + img2 * a).astype(np.uint8)
    return [Image.fromarray(f)] + fade_recursivo(img1, img2, i + 1, n)

img_lena = np.array(Image.open('lena_img.jpeg').convert('RGB'), float)
img_tigre = np.array(Image.open('bengal_tiger.jpeg').convert('RGB'), float)

h = min(img_lena.shape[0], img_tigre.shape[0])
w = min(img_lena.shape[1], img_tigre.shape[1])

frames = fade_recursivo(img_lena[:h, :w], img_tigre[:h, :w])
frames[0].save('animacion_fade.gif', save_all=True, append_images=frames[1:], duration=80, loop=0)

- Para llegar a la imágen objetivo, se cargo la imágen de ejemplo y así tener una presición de los colores de cada cuadro que contiene la imagen final a continuación.

from PIL import Image

img = np.array(Image.open('lena_img.jpeg').convert('RGB'))

c_r = np.array([255, 79, 64])
c_g = np.array([134, 255, 67])
c_b = np.array([129, 76, 255])

h, w, _ = img.shape
r_grid, c_grid = 6, 8

y = (np.arange(h) * r_grid) // h
x = (np.arange(w) * c_grid) // w
yy, xx = np.meshgrid(y, x, indexing='ij')

d = (r_grid - 1 - yy) + xx
color_idx = d % 3

mask = np.zeros_like(img)
mask[color_idx == 0] = c_g
mask[color_idx == 1] = c_b
mask[color_idx == 2] = c_r

res = (img * 0.4 + mask * 0.6)

cv2_imshow(res[:, :, ::-1])
